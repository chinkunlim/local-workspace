### Pseudo-code

```python
# в”Җв”Җ ж–°еўһ helper еҮҪж•ёпјҲp01_transcribe.py й ӮеұӨпјҢclass еӨ–пјүв”Җв”Җ

def vad_preprocess(audio_path: str, tmp_dir: str) -> str:
    """
    з”Ё pydub еҲҮйҷӨйқңйҹіж®өпјҢијёеҮәд№ҫж·Ёзҡ„ WAV дҫӣ Whisper дҪҝз”ЁгҖӮ

    Args:
        audio_path: еҺҹе§Ӣ .m4a и·Ҝеҫ‘
        tmp_dir:    жҡ«еӯҳијёеҮәзӣ®йҢ„пјҲеҰӮ data/audio-transcriber/output/tmp/пјү

    Returns:
        cleaned_wav_path: еҺ»йқңйҹіеҫҢзҡ„ .wav зө•е°Қи·Ҝеҫ‘

    Algorithm:
        1. ијүе…Ҙйҹій »пјҲpydub иҮӘеӢ•е‘јеҸ« ffmpeg и§Јзўј .m4aпјү
        2. д»Ҙ detect_silence() жүҫеҮәжүҖжңүйқңйҹіеҚҖж®ө
        3. дҝқз•ҷйқһйқңйҹізүҮж®өпјҢе…ҒиЁұжңҖеӨҡ 700ms зҡ„йқңйҹідҪңзӮәиҮӘз„¶еҒңй “
        4. жӢјжҺҘжҲҗе–®дёҖ AudioSegment
        5. ијёеҮәзӮә 16kHz mono WAVпјҲWhisper жңҖдҪіж јејҸпјү
        6. иӢҘйқһйқңйҹізүҮж®ө < 1 з§’пјҲе…ЁйқңйҹіжӘ”пјүпјҢзӣҙжҺҘеӣһеӮіеҺҹи·Ҝеҫ‘дёҰиЁҳйҢ„иӯҰе‘Ҡ
    """
    from pydub import AudioSegment
    from pydub.silence import detect_nonsilent

    audio = AudioSegment.from_file(audio_path)           # и§Јзўј .m4a
    audio = audio.set_channels(1).set_frame_rate(16000)  # жЁҷжә–еҢ–зӮә 16kHz mono

    # еҸ–еҫ—йқһйқңйҹіеҚҖй–“еҲ—иЎЁпјҡ[[start_ms, end_ms], ...]
    nonsilent_ranges = detect_nonsilent(
        audio,
        min_silence_len=1500,   # и¶…йҒҺ 1500ms жүҚз®—гҖҢеҖјеҫ—еҲҮйҷӨзҡ„йқңйҹігҖҚ
        silence_thresh=-40,     # dBFS й–ҫеҖјпјҲеҸҜеҫһ config и®ҖеҸ–пјү
        seek_step=10,           # жҗңе°ӢжӯҘйҖІ 10ms
    )

    if not nonsilent_ranges:
        # е…ЁйқңйҹіжӘ”пјҢи·ійҒҺиҷ•зҗҶпјҢи®“ Whisper иҮӘиЎҢиҷ•зҗҶпјҲйҖҡеёёијёеҮәз©әзҷҪпјү
        log("вҡ пёҸ еҒөжё¬еҲ°е…ЁйқңйҹійҹіжӘ”пјҢи·ійҒҺ VAD еүҚиҷ•зҗҶгҖӮ")
        return audio_path

    # жӢјжҺҘйқһйқңйҹізүҮж®өпјҢеңЁзүҮж®өй–“жҸ’е…Ҙ 300ms paddingпјҲйҒҝе…ҚжҲӘй ӯеҺ»е°ҫпјү
    PADDING_MS = 300
    chunks = []
    for start, end in nonsilent_ranges:
        padded_start = max(0, start - PADDING_MS)
        padded_end   = min(len(audio), end + PADDING_MS)
        chunks.append(audio[padded_start:padded_end])

    cleaned_audio = sum(chunks)  # AudioSegment concatenation

    # иЁҲз®—йқңйҹіжҜ”дҫӢдёҰиЁҳйҢ„
    original_duration = len(audio) / 1000
    cleaned_duration  = len(cleaned_audio) / 1000
    silence_ratio     = 1.0 - (cleaned_duration / original_duration)
    log(f"рҹ”Ү VAD еүҚиҷ•зҗҶпјҡз§»йҷӨ {silence_ratio:.1%} йқңйҹіпјҢ{original_duration:.1f}s вҶ’ {cleaned_duration:.1f}s")

    # еҢҜеҮәд№ҫж·Ёзҡ„ WAV
    os.makedirs(tmp_dir, exist_ok=True)
    base_name    = os.path.splitext(os.path.basename(audio_path))[0]
    cleaned_path = os.path.join(tmp_dir, f"{base_name}_vad.wav")
    cleaned_audio.export(cleaned_path, format="wav")

    return cleaned_path
```

## еұӨдәҢпјҡжІ»жЁҷдҝ®еҫ© вҖ” Segment йҮҚиӨҮеҒөжё¬ + еұҖйғЁйҮҚи©Ұ

### йҮҚиӨҮеҒөжё¬жј”з®—жі•иЁӯиЁҲ

еҒөжё¬еҺҹзҗҶпјҡзңҹеҜҰдәәйЎһиӘһйҹізҡ„ N-gram еҲҶеёғжҳҜеӨҡжЁЈзҡ„пјӣе№»иҰәиҝҙеңҲзҡ„еЈ“зё®зҺҮжҘөй«ҳпјҲеҗҢжЁЈзҡ„зүҮиӘһеҸҚиҰҶеҮәзҸҫпјүгҖӮ

```python
def detect_repetition(text: str, ngram_size: int = 4, threshold: float = 0.45) -> bool:
    """
    еҒөжё¬ Segment ж–Үеӯ—жҳҜеҗҰйҷ·е…Ҙе№»иҰәиҝҙеңҲгҖӮ

    Strategy 1 вҖ” N-gram йҮҚиӨҮзҺҮпјҡ
        иЁҲз®—жүҖжңү N-gram дёӯгҖҢйҮҚиӨҮ N-gram дҪ”жҜ”гҖҚгҖӮ
        иӢҘи¶…йҒҺ thresholdпјҢиҰ–зӮәе№»иҰәгҖӮ
        дҫӢпјҡ"и¬қи¬қи§ҖзңӢи¬қи¬қи§ҖзңӢи¬қи¬қи§ҖзңӢ" вҶ’ bigram йҮҚиӨҮзҺҮ вүҲ 0.9

    Strategy 2 вҖ” еЈ“зё®жҜ”е•ҹзҷјејҸ (Compression Ratio Heuristic)пјҡ
        дҪҝз”Ё zlib еЈ“зё®ж–Үеӯ—пјҢиӢҘеЈ“зё®еҫҢеӨ§е°Ҹ / еҺҹе§ӢеӨ§е°Ҹ < 0.3пјҢ
        д»ЈиЎЁж–Үеӯ—й«ҳеәҰйҮҚиӨҮпјҲдҪҺзҶөпјүпјҢиҰ–зӮәе№»иҰәгҖӮ
        пјҲжӯӨжі•дҫҶиҮӘ OpenAI Whisper е®ҳж–№ heuristicпјү

    Args:
        text:       segment зҡ„зҙ”ж–Үеӯ—
        ngram_size: N-gram иҰ–зӘ—еӨ§е°ҸпјҲй җиЁӯ 4-gramпјү
        threshold:  йҮҚиӨҮзҺҮй–ҫеҖјпјҲй җиЁӯ 0.45пјҢи¶…йҒҺеҚіи§ёзҷјпјү

    Returns:
        True  вҶ’ еҒөжё¬еҲ°е№»иҰәиҝҙеңҲпјҢйңҖиҰҒйҮҚи©Ұ
        False вҶ’ жӯЈеёёж–Үеӯ—
    """
    import zlib

    # з©әеӯ—дёІжҲ–йҒҺзҹӯж–Үеӯ—дёҚиҷ•зҗҶ
    words = text.strip().split()
    if len(words) < ngram_size * 2:
        return False

    # Strategy 1: N-gram йҮҚиӨҮзҺҮ
    ngrams = [tuple(words[i:i+ngram_size]) for i in range(len(words) - ngram_size + 1)]
    unique_ngrams = set(ngrams)
    repetition_ratio = 1.0 - (len(unique_ngrams) / len(ngrams))
    if repetition_ratio > threshold:
        return True

    # Strategy 2: zlib еЈ“зё®жҜ”
    encoded     = text.encode("utf-8")
    compressed  = zlib.compress(encoded)
    compress_ratio = len(compressed) / len(encoded)
    if compress_ratio < 0.3:
        return True

    return False
```

### еұҖйғЁйҮҚи©ҰйӮҸијҜ Pseudo-code

```python
def retry_segment(
    audio_path: str,
    start_sec: float,
    end_sec: float,
    tmp_dir: str,
    engine: str,
    model,
    model_name: str,
) -> str:
    """
    йҮқе°Қе№»иҰә SegmentпјҢеҲҮеүІеұҖйғЁйҹіи»ҢпјҢд»Ҙеҡҙж јеҸғж•ёйҮҚж–°иҪүйҢ„гҖӮ

    Args:
        audio_path: еҺҹе§ӢпјҲVAD еҫҢпјүйҹіжӘ”и·Ҝеҫ‘
        start_sec:  з•°еёё Segment й–Ӣе§ӢжҷӮй–“пјҲз§’пјү
        end_sec:    з•°еёё Segment зөҗжқҹжҷӮй–“пјҲз§’пјү
        tmp_dir:    жҡ«еӯҳзӣ®йҢ„
        engine:     "mlx-whisper" жҲ– "faster-whisper"
        model:      е·Іијүе…Ҙзҡ„ Whisper model зү©д»¶
        model_name: model еҗҚзЁұпјҲдҫӣ mlx-whisper дҪҝз”Ёпјү

    Returns:
        йҮҚж–°иҪүйҢ„зҡ„ж–Үеӯ—пјҲиӢҘйҮҚи©ҰеҫҢд»Қз•°еёёпјҢеӣһеӮі "[иҪүйҢ„еӨұж•—]" дҪ”дҪҚз¬Ұпјү
    """
    from pydub import AudioSegment

    # еҲҮеүІеұҖйғЁйҹіи»ҢпјҲеүҚеҫҢеҗ„еҠ  200ms bufferпјү
    BUFFER_MS = 200
    audio     = AudioSegment.from_file(audio_path)
    start_ms  = max(0, int(start_sec * 1000) - BUFFER_MS)
    end_ms    = min(len(audio), int(end_sec * 1000) + BUFFER_MS)
    clip      = audio[start_ms:end_ms]

    clip_path = os.path.join(tmp_dir, f"retry_{int(start_sec)}_{int(end_sec)}.wav")
    clip.export(clip_path, format="wav")

    # д»Ҙеҡҙж јеҸғж•ёйҮҚи©Ұ
    try:
        if engine == "mlx-whisper":
            import mlx_whisper
            result = mlx_whisper.transcribe(
                clip_path,
                path_or_hf_repo=model_name,
                condition_on_previous_text=False,  # вҶҗ й—ңйҚөйҳІзҰҰ
                temperature=0.2,                   # вҶҗ иј•еҫ®йҡЁж©ҹжҖ§жү“з ҙиҝҙеңҲ
                verbose=False,
            )
            retry_text = " ".join(s["text"].strip() for s in result.get("segments", []))

        elif engine == "faster-whisper":
            segments_gen, _ = model.transcribe(
                clip_path,
                condition_on_previous_text=False,  # вҶҗ й—ңйҚөйҳІзҰҰ
                temperature=0.2,
                vad_filter=True,
            )
            retry_text = " ".join(s.text.strip() for s in segments_gen)

        # еҶҚж¬ЎзўәиӘҚйҮҚи©ҰзөҗжһңжҳҜеҗҰд»Қжңүе№»иҰә
        if detect_repetition(retry_text):
            log(f"вҡ пёҸ Segment [{start_sec:.1f}s-{end_sec:.1f}s] йҮҚи©ҰеҫҢд»Қз•°еёёпјҢжЁҷиЁҳзӮә [иҪүйҢ„еӨұж•—]")
            return "[иҪүйҢ„еӨұж•—]"

        return retry_text

    except Exception as e:
        log(f"вқҢ Segment йҮҚи©ҰеӨұж•—: {e}", "error")
        return "[иҪүйҢ„еӨұж•—]"
    finally:
        # жё…зҗҶжҡ«еӯҳ
        if os.path.exists(clip_path):
            os.remove(clip_path)
```
